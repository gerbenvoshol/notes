Change to directory if possible

cd $DIR || {
  echo "Cannot change to necessary directory." >&2
  exit $E_XCD;
}

Iterate over files

FILES="/var/log/*.log"
for file in $FILES
do
  echo "$file"
done
# Using ls (here sorted after ctime)
FILES=$( ls -c /var/log/*.log )
for file in $FILES
do
  echo $file
done
